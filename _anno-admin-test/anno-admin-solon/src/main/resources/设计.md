沿用 anno-admin相关设计

采用通用的XML格式


```xml

<class bind="site.sorghum.anno.test.modular.better.Article">
    <fields>
        <field bind="id"
               title="编号"
               tableField="id"
               type="site.sorghum.anno.anno.enums.AnnoDataType"/>
        <field bind="title"
               title="标题"
               type="site.sorghum.anno.anno.enums.AnnoDataType"/>
    </fields>
    <searchs>
        <search bind="title" required="true"/>
    </searchs>
    <edits>
        <edit bind="title" required="true"/>
    </edits>
    <adds>
        <add bind="title" required="true"/>
    </adds>
</class>
```
